// =================== Анкеты ===================

Table Survey {
  id int [primary key]
  userId int
  name varchar [not null]
  description varchar
  isPublished bool [default: false]
  access enum [default: "private"]
  totalPoints int
  createdAt timestamp [not null]
  updatedAt timestamp [not null]
}

Table Question {
  id int [primary key]
  surveyId int [not null]
  name varchar
  page int [not null]
  position int [default: 1]
  questionText varchar
  isMandatory boolean [default: false]
  answer varchar
  points int [default: 0]
  type enum [not null]
}

Table QuestionOption {
  id int [primary key]
  questionId int [not null]
  position int [not null]
  isCorrect bool [default: false]
  points int [default: 0]
  text varchar [not null]
}

// =================== Ответы ===================

Table Response {
  id int [primary key]
  userId int
  surveyId int [not null]
  score int [default: 0]
  totalPoints int
  createdAt timestamp [not null]
}

Table Answer {
  id int [primary key]
  // userId int [unique, not null]
  responseId int [not null]
  questionId int [not null]
  answerText varchar
}

Table AnswerOption {
  id int [primary key]
  optionId int [not null]
  answerId int [not null]
}

// =============== Пользователи ================

Table User {
  id int [primary key]
  login varchar [unique, not null]
  password varchar [not null]
  createdAt timestamp [not null]
  updatedAt timestamp [not null]
}

Table RefreshSession {
  id int [primary key]
  userId int [not null]
  refreshToken varchar [unique, not null]
  userAgent varchar
  ip varchar
  fingerprint varchar
  expiresAt timestamp [not null]
  createdAt timestamp [not null]
}

// =================== Связи ===================

Ref: User.id < Survey.userId
Ref: User.id < Response.userId
Ref: Survey.id < Question.surveyId
Ref: Question.id < QuestionOption.questionId

// Ref: User.id < Answer.userId
Ref: Survey.id < Response.surveyId
Ref: Response.id < Answer.responseId
Ref: Question.id < Answer.questionId
Ref: AnswerOption.answerId > Answer.id
Ref: AnswerOption.optionId > QuestionOption.id

Ref: User.id < RefreshSession.userId
